Installazione in /opt/python

~/bin/install_python.sh

#!/usr/bin/env bash
VERSION=3.9.6
EXT=tgz
PYTHONFTP=https://www.python.org/ftp/python
PYTHONTAR=Python-$VERSION.$EXT
PYTHONDIR=${PYTHONTAR%.$EXT}
CHM=python${VERSION//\./}.chm
DST=/opt/python
WORKDIR=~/workspaces/Python/python/
COMPILE=true
sudo rm -rf $DST
sudo mkdir -p $DST
sudo chown -R $USER:$USER $DST
# https://stackoverflow.com/questions/2953646/how-can-i-declare-and-use-boolean-variables-in-a-shell-script
# https://stackoverflow.com/questions/2953646/how-can-i-declare-and-use-boolean-variables-in-a-shell-script/21210966#21210966
if [ "$COMPILE" = true ]; then
        rm -rf $WORKDIR
        mkdir -p $WORKDIR
        cd $WORKDIR
        sudo pacman -Syu --needed base-devel
        wget $PYTHONFTP/$VERSION/$PYTHONTAR
        wget $PYTHONFTP/$VERSION/$CHM
        tar xvfz $PYTHONTAR
        cd $PYTHONDIR
        #./configure --help # per avere le opzioni di compilazione
        ./configure --enable-optimizations --prefix $DST
        make -j # If the -j option is given without an argument, make will not limit the number of jobs that can run simultaneously.
        # make -j$(nproc) # nproc ritorna il numero dei core
        # make test
else
        cd $WORKDIR/$PYTHONDIR
fi
make install
# Creazione link python e pip che verranno posti in cima al PATH
ln -s $DST/bin/python3.9 $DST/bin/python
ln -s $DST/bin/pip3 $DST/bin/pip
# Aggiunta del nuovo python in cima al PATH mediante .bashrc
# head -c -1 elimina l'ultimo newline
cat <<EOF| head -c -1 >> ~/.bashrc
# Python
export PATH=$DST/bin:
EOF
cat <<'EOF' >> ~/.bashrc
$PATH
EOF
cd
. ~/.bashrc
which python
python --version
pip install pip install jupyterlab ipython
