# Creazione di un nuovo ramo, modifiche e push sul server
git checkout -b 1.5
git status
git commit -a -m "I nostri servizi"
git tag -a 1.5.1 -m "I nostri servizi"
git push -u origin 1.5
git push -f --tags

# Si vogliono accorpare tutti i commit del nuovo ramo in un solo commit e reinviare al server:

$ git reflog # serve a trovare il commit antecedente al primo commit effettuato sul nuovo ramo
$ git reset --soft HEAD@{6} # si sposta l'HEAD alla reference del commit precedente alla creazione del ramo, senza toccare index e working directory (--soft), reset su commit rende orfani i commit seguenti alla nuova posizione dell'head
$ git commit -a -m "Aggiornamento I nostri servizi" # si crea un nuovo commit che accorpa tutti i commit precedenti del ramo (resi orfani da reset --soft), che vengono poi eliminati dal garbage collector di git
$ git tag -fa 1.5.1 -m "Aggiornamento I nostri servizi" # si sposta il tag sul nuovo commit
$ git push origin +1.5 # si invia al server solamente il nuovo ramo, sintassi + al posto di --force
$ git push -f --tags   # si invia il tag spostato al server
$ git log --oneline --graph --decorate
