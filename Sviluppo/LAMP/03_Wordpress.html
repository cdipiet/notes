NON bisogna installare wordpress con pacman ma con il download dal sito.
Prima bisogna aver installato e configurato:

  1. MariaDB
  2. Apache HTTP Server
  3. PHP
  4. PHPMyAdmin
# Installare Wordpress
$ sudo pacman -Syuu --noconfirm
$ sudo su
$ cd /srv/http/
$ wget https://wordpress.org/latest.tar.gz
$ # curl -O  https://wordpress.org/latest.tar.gz
# Viene creata la cartella /srv/http/wordpress
$ tar xvzf latest.tar.gz
$ sudo chown -R http:http /srv/http/wordpress

$ sudo bash -c "cat > /etc/httpd/conf/extra/wordpress.conf" << 'EOF'
Alias /wordpress "/srv/http/wordpress"
<Directory "/srv/http/wordpress">
    DirectoryIndex index.php
    AllowOverride All
    Options FollowSymlinks
    Require all granted
</Directory>
EOF
$ # sudo code --no-sandbox --user-data-dir=/root/.config/Code /etc/httpd/conf/httpd.conf
$ sudo bash -c "cat >> /etc/httpd/conf/httpd.conf" << 'EOF'
Include conf/extra/wordpress.conf
<IfModule dir_module>
    DirectoryIndex index.php
</IfModule>
EOF
$ apachectl configtest
$ sudo systemctl restart httpd

# MariaDB Wordpress
$ sudo bash -c "mariadb -u root -p" << EOF
DROP DATABASE IF EXISTS wordpress;
CREATE DATABASE wordpress;
DROP USER 'wp-user'@'localhost';
CREATE USER 'wp-user'@'localhost' IDENTIFIED BY 'wp-user';
GRANT ALL PRIVILEGES ON wordpress.* TO 'wp-user'@'localhost' WITH GRANT OPTION;
FLUSH PRIVILEGES;
EOF
$ sudo systemctl restart mariadb
$ sudo systemctl restart httpd
