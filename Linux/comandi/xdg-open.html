xdg-open - opens a file or URL in the user's preferred application

ft=$(xdg-mime query filetype "mypage.html")
echo "File Type: " $ft
aa=$(xdg-mime query default $ft)
echo "Associated Application: " $aa
xdg-open "mypage.html"
