Per file multipli ðŸ”—

$ cat << 'EOF' >> fsichk.conf
/bin/*
/boot/vmlinuz*
/etc/fstab
/usr/sbin/*
EOF

$ cat << 'EOF' > makesums
#!/bin/bash
CONFIG_FILE=./fsichk.conf
CHECKSUM_FILE=./checksums
for i in `cat $CONFIG_FILE`; do
if [ ! -d $i ]; then
    sudo bash -c "md5sum $i" >> $CHECKSUM_FILE
fi
done
EOF
$ chmod +x makesums
$ ./makesums
# I checksum generati saranno salvati nel file indicato da CHECKSUM_FILE.
# Per ricontrollare tutti i checksum salvati nel file CHECKSUM_FILE:

$ sudo md5sum -c ./checksums
