Impariamo awk ðŸ”—

Alfred Aho, Peter Weinberger, and Brian Kernighan

Esempi di espressioni di selezione riga ðŸ”—

  • $1 == "root" seleziona le righe il cui primo campo vale "root"
  • BEGIN l'azione associata a questo pattern viene eseguita prima di leggere la prima riga dell'input
  • END l'azione associata a questo pattern viene eseguita dopo che sia stata letta l'ultima riga dell'input
  • /^[0-9]/ seleziona le righe che iniziano con una cifra
  • $3 < 1000 && /bash$/ esempio di and tra un confronto su un campo ed una espressione regolare
  • NR=5,/^nobody/ i pattern separati da virgola costituiscono un intervallo di righe. In questo esempio dalla riga 5 a quella che inizia per "nobody"

$0 rappresenta l'intera riga

    $ df | awk '/^\/dev/ {print $1 ": "$5}' # Seleziona le righe che iniziano per /dev e stampa il primo e quinto campo

Esempio di programma awk in un file ðŸ”—

maxuid.awk

BEGIN {FS=":"; maxuid=0}
$3 > maxuid {maxuid=$3;maxname=$1}
END {print maxname ":" maxuid}

$ awk -f maxuid.awk /etc/passwd

Esempio di programma awk come script con shebang ðŸ”—

maxuid.awk

#!/usr/bin/awk -f
BEGIN {FS=":"; maxuid=0}
$3 > maxuid {maxuid=$3;maxname=$1}
END {print maxname ":" maxuid}

$ chmod +x maxuid.awk
$ ./maxuid.awk /etc/passwd

Il pattern o l'azione possono mancare nella regola, se manca il pattern la regola viene eseguita sempre,
se manca l'azione la riga viene stampata quando il pattern e' verificato:

awk -F: '$3>=1000' /etc/passwd # mostra tutti gli account con UID >= 1000
awk -F: '{print $1}' /etc/passwd # stampa i nomi di tutti gli account

-F come opzione di lancio oppure FS come comando, specificano il separatore di campo

Esempi ðŸ”—

Stampare ogni riga con numero multiplo di 10 ðŸ”—

NR%10 == 0

Stampare le righe numerate ðŸ”—

{print NR, $0}

Stampare tutte le righe lette in ordine inverso ðŸ”—

{s=$0 "\n" s} END {print s}

Stampare le righe centrate orizzontalmente, colonna 40 ðŸ”—

{printf "%" int(40+length($0)/2) "s\n", $0}

Stampare solo le righe non vuote ðŸ”—

NF

NF viene aggiornata ad ogni riga e contiene il numero dei campi

Stampare righe piu lunghe di 80 caratteri ðŸ”—

length($0) > 80

Sostituisce TTY con TERMINAL solo nella prima riga ðŸ”—

NR==1 {gsub("TTY","TERMINAL"); print} NR!=1

la seconda condizione NR!=1 permette di stampare inalterate tutte le righe successive alla prima

Elaborare record su piu righe ðŸ”—

The empty string, "" (a string of no characters), has a special meaning as the value of RS: it means that records are separated only by blank lines

Si cambia il record separator RS. ponendo RS="" si definisce come separatore di record una riga vuota, e ponendo FS="\n" si definisce il separatore di campo come il newline.

Esempio file di input, triplette nome_marito,citta,nome_moglie:

Homer Simpson
Springfield
Marge

Fred Flintstone
Bedrock
Wilma

BEGIN {RS=""; FS="\n"}
$2 == "Bedrock" {print $3}

totmem.awk ðŸ”—

Conteggio della memoria utilizzata da ogni utente, il primo campo di ps aux e' il nome dell'utente mentre il quinto, VSZ e' la dimensione della memoria virtuale allocata al processo. Usiamo due array associativi, "numero" conterra il numero totale dei processi dell'utente, "totale" conterra il totale della memoria utilizzata dall'utente.

cdipiet@debian:~$ ps aux
USER PID %CPU %MEM VSZ RSS TTY STAT START TIME COMMAND
root 1 0.0 0.0 10448 772 ? Ss 07:41 0:01 init [2]
root 2 0.0 0.0 0 0 ? S 07:41 0:00 [kthreadd]
cdipiet 2473 0.0 0.4 402612 39424 ? Rl 07:42 0:14 /usr/bin/yakuake
cdipiet 2781 0.1 0.4 191808 32748 ? S 07:43 1:02 emerald --replace

$1 != "USER" { numero[$1]++; totale[$1] += $5 }
END {
for (utente in totale)
printf "%8s\t%4d\t%8d\n", utente, numero[utente], totale[utente]
}

$ ps aux|awk -f totmem.awk

#!/bin/awk -f
BEGIN {

# Print the squares from 1 to 10 the first way

i=1;
while (i <= 10) {
printf "The square of ", i, " is ", i*i;
i = i+1;
}

# do it again, using more concise code

for (i=1; i <= 10; i++) {
printf "The square of ", i, " is ", i*i;
}

# now end
exit;
}

#!/usr/bin/env bash
find . -type f -print | xargs /usr/bin/ls -islg |
awk '
BEGIN {
# initialize all arrays used in for loop
u_count[""]=0;
g_count[""]=0;
ug_count[""]=0;
all_count[""]=0;
}
{
# validate your input
if (NF != 11) {
# ignore
} else {
# assign field names
inode=$1;
size=$2;
linkcount=$4;
user=$5;
group=$6;

# should I count this file?

doit=0;
if (linkcount == 1) {
# only one copy - count it
doit++;
} else {
# a hard link - only count first one
seen[inode]++;
if (seen[inode] == 1) {
doit++;
}
}
# if doit is true, then count the file
if (doit ) {

# total up counts in one pass
# use description array names
# use array index that unifies the arrays

# first the counts for the number of files

u_count[user " *"]++;
g_count["* " group]++;
ug_count[user " " group]++;
all_count["* *"]++;

# then the total disk space used

u_size[user " *"]+=size;
g_size["* " group]+=size;
ug_size[user " " group]+=size;
all_size["* *"]+=size;
}
}
}
END {
# output in a form that can be sorted
for (i in u_count) {
if (i != "") {
print u_size[i], u_count[i], i;
}
}
for (i in g_count) {
if (i != "") {
print g_size[i], g_count[i], i;
}
}
for (i in ug_count) {
if (i != "") {
print ug_size[i], ug_count[i], i;
}
}
for (i in all_count) {
if (i != "") {
print all_size[i], all_count[i], i;
}
}
} ' |
# numeric sort - biggest numbers first
# sort fields 0 and 1 first (sort starts with 0)
# followed by dictionary sort on fields 2 + 3
