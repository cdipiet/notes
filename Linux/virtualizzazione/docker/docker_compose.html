Disabilitare un servizio ðŸ”—

Per disabilitare un servizio lo si puo inserire in un profilo, solamente i servizi senza profilo vengono eseguiti di default.

  • https://docs.docker.com/compose/reference/run/
  • https://docs.docker.com/compose/profiles/

il servizio Ã¨ sempre eseguibile, specificando il profilo assegnato in docker-compose up

compose.yml

version: "3.9"
services:
  frontend:
    image: frontend
    profiles: ["frontend"]

  phpmyadmin:
    image: phpmyadmin
    depends_on:
      - db
    profiles:
      - debug

  backend:
    image: backend

  db:
    image: mysql

$ docker-compose --profile debug up

Dockerfile

FROM node:14
WORKDIR /usr/src/app
COPY package.json .
RUN npm install
COPY . .
EXPOSE 3000
CMD ["node", "server.js"]

docker-compose.yml

version: 3.8
services:
    mongo:
        image: mongo:4.4
        container_name: docker_mongo
        ports:
            - "27017:27017"
    app:
        build: ./app
        container_name: node_app
        ports:
            - "3000:3000"
        environment:
            - MONGO_URL=mongodb://mongo:27017/testdb
        depends_on:
            - mongo

$ docker compose up

version: 3.8
services:
    build:
        image: node:16
        volumes:
            - ./app:/app
        working_dir: /app
        command: sh -c "npm install && npm run build"
    nginx:
        image: nginx:1.21
        depends_on:
            - build
        volumes:
            - ./app/build:/usr/share/nginx/html:ro
        ports:
            - "8000:80"
