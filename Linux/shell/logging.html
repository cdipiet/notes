#!/usr/bin/env bash
function logit(){

    local LOG_LEVEL=$1
    shift
    MSG=$@
    # TIMESTAMP=$(date +"%F")           # 2020-08-16
    TIMESTAMP=$(date +"%Y-%m-%d %T")    # 2020-08-16 13:01:25
    if [ LOG_LEVEL = 'ERROR' ] || $VERBOSE
    then
    echo "${TIMESTAMP} ${HOST} ${PROGRAM_NAME} [${PID}]: ${LOG_LEVEL} ${MSG}"
    fi
}
