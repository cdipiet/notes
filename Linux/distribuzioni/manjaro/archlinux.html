lsblk
# Per vedere se siamo in un sistema UEFI
ls /sys/firmware/efi/efivars
blkid /dev/sda1 # per cercare la partizione UEFI

# Per resettare e creare partizioni GPT UEFI:
gdisk /dev/sda
    o       # reset tabella delle partizioni
    n       # nuova partizione
    1       # partition number
    invio   # settore iniziale
    200M    # per creare una nuova partizione UEFI
    ef00    # per creare una nuova partizione UEFI, L per cercare i codici delle partizioni   
            # 8300 per creare una partizione Linux
    w       # write
lsblk

mkfs.fat -F32 /dev/sda1 # per creare una nuova partizione UEFI

# Creare un btrfs per /root
mkfs.btrfs -f /dev/sda2
blkid /dev/sda2
mount /dev/sda2 /mnt
cd /mnt
brtfs subvolume create @        # crea il subvolume per root
brtfs subvolume create @home
cd
umount /mnt
mount -o noatime,ssd,space_cache=v2,compress=zstd,discard=async,subvol=@ /dev/sda2 /mnt
mkdir /mnt/home
mount -o noatime,ssd,space_cache=v2,compress=zstd,discard=async,subvol=@home /dev/sda2 /mnt/home

# montiamo anche la partizione efi, che puÃ² anche giÃ  esistere perchÃ¨ creata da windows:
mkdir -p /mnt/boot/efi 
mount /dev/sda1 /mnt/boot/efi

# Per vedere anche la partizione windows, es. /dev/sdb1
mkdir /mnt/windows
lsblk
mount /dev/sdb1 /mnt/windows

# per cercare i mirror piÃ¹ veloci, 6 ore di anzianitÃ 
reflector --country Italy --age 6 --sort rate --save &etc/pacman.d/mirrorlist
pacman -Sy

# utilizzato per le nuove installazioni, -K installa anche un keyring
# per amd c'e' un altro microcode invece di intel-ucode
pacstrap -K /mnt base linux linux-firmware intel-ucode btrfs-progs vim

# Per generare in automatico /etc/fstab c'e' una utility che prende le info dalle partizioni che abbiamo appena montato
genfstab -U /mnt >> /mnt/etc/fstab
cat /mnt/etc/fstab

# Spostiamoci ora nella nuova installazione
arch-chroot /mnt
ln -sf /usr/share/zoneinfo/Europe/Rome /etc/localtime
# sincronizziamo l'hardware clock con il system clock
hwclock --systohc
vim /etc/locale.gen
    /it_IT
    i  e rimuovere il commento

        arch
passwd # cambiare la password di root del nuovo sistema(non di quello in cui siamo entrati tramite ssh)

base-devel linux-headers bluez bluez-utils cups hplip alsa-utils pipewire pipewire-alsa pipewire-pulse pipewire-jack
bash-completion openssh rsync acpi acpi_call tlp sof-firmware acpid os-prober ntfs-3g nvidia

    uncomment:
    GRUB_DISABLE_OS_PROBER=false
