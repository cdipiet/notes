Dove si trova la directory di build ðŸ”—

Correggere il checksum con sed ðŸ”—

# Per testare l'espressione regolare:
$ cat PKGBUILD | sed "s/^sha256sums=.*$/$sums/"
$ cat PKGBUILD | sed "s/^sha256sums=.*$/$sums/" > PKGBUILD.test
$ code -d PKGBUILD PKGBUILD.test

mg() {
        if [ ! -f ./PKGBUILD ]; then
                echo "No PKGBUILD in `pwd`"
                return 1
        fi

        local sums
        # calculate new checksums
        sums=$(makepkg -g)
        # replace them in-place
        sed -i "s/^sha256sums=.*$/$sums/" PKGBUILD
        echo "PKGBUILD updated"
}
